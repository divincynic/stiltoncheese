(function(){ 'use strict'; const canvas = document.getElementById('c'); const renderer = new THREE.WebGLRenderer({canvas, antialias:true}); renderer.setPixelRatio(Math.min(window.devicePixelRatio,2)); renderer.setSize(window.innerWidth, window.innerHeight); renderer.shadowMap.enabled = true; renderer.shadowMap.type = THREE.PCFSoftShadowMap; renderer.toneMapping = THREE.ACESFilmicToneMapping; renderer.toneMappingExposure = 1.05; renderer.outputEncoding = THREE.sRGBEncoding; const scene = new THREE.Scene(); scene.background = new THREE.Color(0x120f0a); scene.fog = new THREE.Fog(0x120f0a, 12, 28); const camera = new THREE.PerspectiveCamera(40, window.innerWidth/window.innerHeight, 0.1, 100); camera.position.set(5.2, 4.2, 6.0); camera.lookAt(0, 0.4, 0); // ── Lighting ─────────────────────────────────────────────────── scene.add(new THREE.AmbientLight(0xfff0d8, 0.28)); const sun = new THREE.DirectionalLight(0xfffae8, 2.4); sun.position.set(5, 9, 4); sun.castShadow = true; scene.add(sun); scene.add(new THREE.DirectionalLight(0xc8dff8, 0.45).position.set(-6, 2, -2)); // ── Table ────────────────────────────────────────────────────── const tableGeo = new THREE.PlaneGeometry(40,40); const tableMat = new THREE.MeshStandardMaterial({color:0x1e1508, roughness:0.97}); const table = new THREE.Mesh(tableGeo, tableMat); table.rotation.x = -Math.PI/2; table.position.y = -0.86; table.receiveShadow = true; scene.add(table); // ── Constants for the Vein System ────────────────────────────── const R = 2.3, HH = 1.75, ANGLE = Math.PI/3; let bodyMesh; const cheeseGroup = new THREE.Group(); scene.add(cheeseGroup); // ── Materials ────────────────────────────────────────────────── const trunkMat = new THREE.MeshStandardMaterial({ color:0x1a5060, emissive:0x061418, emissiveIntensity:0.6, roughness:0.3, metalness:0.1 }); // ── Needle Function (RESTORED) ───────────────────────────────── function flashNeedle(worldPt){ const geo = new THREE.CylinderGeometry(0.007, 0.003, 2.8, 6); const mat = new THREE.MeshStandardMaterial({color:0xc8d4dc, metalness:0.9, roughness:0.1}); const n = new THREE.Mesh(geo, mat); n.position.copy(worldPt); n.position.y += 1.5; scene.add(n); let f = 0; function step(){ f++; if(f < 15) n.position.y -= 0.12; // Down else if(f < 30) n.position.y += 0.12; // Up else { scene.remove(n); geo.dispose(); mat.dispose(); return; } requestAnimationFrame(step); } step(); } // ── Model Loader ─────────────────────────────────────────────── const loader = new THREE.GLTFLoader(); loader.load('models/cheese.glb', (gltf) => { const model = gltf.scene; model.traverse((child) => { if (child.isMesh) { child.castShadow = true; child.receiveShadow = true; // Note: We use your existing pasteMat/rindMat from your previous code here if (child.name.toLowerCase().includes('rind')) child.material = rindMat; else { child.material = pasteMat; bodyMesh = child; } } }); cheeseGroup.add(model); }); // ── Vein Spawning ───────────────────────────────────────────── let veinCount = 0; let pierceCount = 0; const veinAnims = []; function spawnVein(localPt){ const grp = new THREE.Group(); // Start the vein slightly inside the click point const origin = localPt.clone().multiplyScalar(0.1); const pts = [origin]; const steps = 8; for(let i=1; i<=steps; i++){ const t = i/steps; const p = origin.clone().lerp(localPt, t); // Add some "wiggle" p.x += (Math.random()-0.5)*0.3; p.y += (Math.random()-0.5)*0.3; pts.push(p); } const curve = new THREE.CatmullRomCurve3(pts); const geo = new THREE.TubeGeometry(curve, 16, 0.025, 6, false); const vMesh = new THREE.Mesh(geo, trunkMat); grp.add(vMesh); grp.scale.setScalar(0.01); // Start tiny cheeseGroup.add(grp); veinAnims.push({grp, s:0.01, target:1.0, done:false}); veinCount++; } // ── Interaction ─────────────────────────────────────────────── const rc = new THREE.Raycaster(); const mouse = new THREE.Vector2(); document.addEventListener('pointerdown', (e) => { if(!bodyMesh) return; const rect = renderer.domElement.getBoundingClientRect(); mouse.x = ((e.clientX - rect.left) / rect.width) * 2 - 1; mouse.y = -((e.clientY - rect.top) / rect.height) * 2 + 1; rc.setFromCamera(mouse, camera); const hits = rc.intersectObject(bodyMesh); if(hits.length > 0){ const wp = hits[0].point; const lp = cheeseGroup.worldToLocal(wp.clone()); flashNeedle(wp); // Show the needle animation spawnVein(lp); // Grow the blue vein pierceCount++; document.getElementById('pv').textContent = pierceCount; document.getElementById('vv').textContent = veinCount; } }); // ── Animation Loop ──────────────────────────────────────────── const clk = new THREE.Clock(); function animate(){ requestAnimationFrame(animate); const dt = clk.getDelta(); // Rotate the cheese slightly cheeseGroup.rotation.y += dt * 0.1; for(const v of veinAnims){ if(!v.done){ v.s = Math.min(v.target, v.s + dt * 2.0); v.grp.scale.setScalar(v.s); if(v.s >= v.target) v.done = true; } } renderer.render(scene, camera); } animate(); })();
